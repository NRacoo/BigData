Outlier¶

In [94]:
cols = ['Age', 'Fare', 'Pclass']
for col in cols:
    Q1 = df[col].quantile(0.25)
    Q3 = df[col].quantile(0.75)

    IQR = Q3 - Q1
    lower = Q1 - 1.5 * IQR
    upper = Q3 + 1.5 * IQR

    outlier = df[(df[col] < lower) | (df[col] > upper)]
    plt.boxplot(df[col])
    plt.show()
